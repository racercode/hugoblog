#include <iostream>
#include <vector>
#define lowbit(x) (x & -x)
#define endl '\n'
#define iofaster ios::sync_with_stdio(0);cin.tie(0);
#define int long long
#define pb push_back
#define pii pair<int, int>
using namespace std;
const int maxn = 5e5 + 5;
const int mod = 1e9 + 7;
const int inf = 1e18 + 9;
const int mxn = 1e9;
const double eps = 1e-7;
int bit[maxn], n, r, k;
vector<pii> v;
void clear() {
    while(!v.empty()) {
        auto [i, c] = v.back();
        bit[i] -= c;
        v.pop_back();
    }
}
inline bool check(int f) {
    if(n <= 2 * r + 1) {
        int mx = 0, sum = 0;
        for(int i = 1; i <= min(r + 1, n); i++) sum += bit[i];
        for(int i = 1; i <= n; i++) {
            mx = max(mx, f - sum);
            if(i - r >= 1) sum -= bit[i - r];
            if(i + r + 1 <= n) sum += bit[i + r + 1];
        }
        return mx <= k;
    }
    else {
        int tot = 0, a = r + 1, b = n - r, sum = 0;
        for(int i = max(1LL, a - 2 * r); i <= a; i++) sum += bit[i];
        for(int i = a; i <= b; i++) {
            int c = f - sum;
            if(c > k) {
                clear();
                return false;
            }
            if(c > 0) {
                tot += c;
                bit[i] += c;
                sum += c;
                v.pb({i, c});
            }
            if(tot > k) {
                clear();
                return false;
            }
            if(a - 2 * r >= 1) sum -= bit[i - 2 * r];
            sum += bit[i + 1];
        }
        sum = 0;
        for(int i = min(b + 2 * r, n); i >= b; i--) sum += bit[i];
        for(int i = b; i >= a; i--) {
            int c = f - sum;
            if(c > k) {
                clear();
                return false;
            }
            if(c > 0) {
                tot += c;
                bit[i] += c;
                sum += c;
                v.pb({i, c});
            }
            if(tot > k) {
                clear();
                return false;
            }
            if(i + 2 * r <= n) sum -= bit[i + 2 * r];
            sum += bit[i - 1];
        }
        clear();
        return tot <= k;
    }
}
signed main() {
    iofaster;
    cin >> n >> r >> k;
    for(int i = 1; i <= n; i++) {
        int x;
        cin >> x;
        bit[i] += x;
    }
    int l = 0, r = 2e18, best;
    while(l <= r) {
        int mid = (l + r) >> 1;
        bool f = check(mid);
        if(f) best = mid, l = mid + 1;
        else r = mid - 1;
    }
    cout << best << endl;
}
