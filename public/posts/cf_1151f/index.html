const int maxn = 1e5 + 5;
const int mod = 1e9 + 7;

int num[maxn], pa[maxn], sz[maxn], arr[maxn], act[maxn], cp[maxn], sz2[maxn], mp[maxn], cnt, n, ct;
vector<int> g[maxn];
vector<int> v[maxn];
int findset(int x) {
    return x == pa[x] ? x : pa[x] = findset(pa[x]);
}
int findset2(int x) {
    return x == cp[x] ? x : cp[x] = findset2(cp[x]);
}
void merge(int a, int b) {
    if(a <= 0 or b > n or act[a] == 0 or act[b] == 0) return ;
    a = findset(a), b = findset(b);
    if(a == b) return ;
    if(sz[a] > sz[b]) swap(a, b);
    pa[a] = b;
    sz[b] += sz[a];
    cnt--;
}
void merge2(int a, int b, int x) {
    if(a <= 0 or b > n or num[a] != x or num[b] != x) return ;
    a = findset2(a), b = findset2(b);
    if(a == b) return ;
    if(sz2[a] > sz2[b]) swap(a, b);
    cp[a] = b;
    sz2[b] += sz2[a];
    ct--;
}
inline int cal(int x) {
    ct = 0;
    for(int u: v[x]) {
        ct++;
        merge2(u - 1, u, x);
        merge2(u, u + 1, x);
    }
    return ct;
}
signed main() {
    int ans = 0, sum = 0;
    cin >> n;
    for(int i = 1; i <= n; i++) pa[i] = i, sz[i] = 1, cp[i] = i, sz2[i] = 1;
    for(int i = 1; i <= n; i++) {
        cin >> num[i];
        v[num[i]].pb(i);
    }
    for(int i = 1; i <= n; i++) {
        if(i != 1) {
            g[num[i]].pb(num[i - 1]);
            if(abs(num[i] - num[i - 1]) == 1) mp[min(num[i - 1], num[i])]++;
        }
        if(i != n) g[num[i]].pb(num[i + 1]);
    }
    for(int i = 1; i <= n; i++) {
        for(int u: v[i]) {
            act[u] = true;
            cnt++;
            merge(u - 1, u), merge(u, u + 1);
        }
        arr[i] = cnt;
        sum += cnt;
        ans += cnt;
    }
    for(int i = 2; i <= n; i++) {
        int r = cal(i - 1);
        int tmp = sum - r - r * (n - i + 1);
        for(int c: g[i - 1]) {
            if(c > i - 1) tmp += n - c + 1;
        }
        ans += tmp;
        sum = tmp;
    }
    cout << ans << endl;
}
